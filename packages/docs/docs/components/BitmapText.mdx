## Usage

```jsx live noInline

  anchor: 0.5,
  x: 250,
  y: 150,
};

render(
    <Stage width={500} height={300} options={{ backgroundColor: 0xffffff }}>
        <PIXILoader
            name="desyrel"
            url="https://s3-us-west-2.amazonaws.com/s.cdpn.io/693612/bitmapfont.xml"
            loader={<Text {...position} text="âŒ› Loading font..." style={{ fontFamily: 'Arial', fontSize: 15 }} />}
        >
            <BitmapText {...position} text="Hello World!" style={{ fontName: 'Desyrel', fontSize: 50 }} />
        </PIXILoader>
    </Stage>
);
```

## Example

Make sure to load the bitmap font data before mounting.

See https://pixijs.io/examples/#/demos/text-demo.js

```js
import { render } from 'react-dom'
import { Stage, BitmapText } from '@pixi/react-pixi'

PIXI.Loader.shared
  .add('desyrel', './assets/desyrel.xml')
  .load(onAssetsLoaded)

const App = () => (
  <Stage>
    <BitmapText text="Hello World" style={{ fontName: 'Desyrel', fontSize: 35 }}>
  </Stage>
)

function onAssetsLoaded() {
  render(<App />, document.getElementById('root'))
}
```
